```javascript
const bcrypt = require("bcryptjs");

const userSchema = new mongoose.Schema({
  email: {
    type: String,
    required: true,
    unique: true,
    lowercase: true,
    trim: true
  },
  password: {
    type: String,
    required: true,
    minlength: 6
  },
  firstName: String,
  lastName: String,
  businessId: mongoose.Schema.Types.ObjectId,
  role: {
    type: String,
    enum: ["owner", "team", "admin"],
    default: "owner"
  },

    type: String,
    enum: ["trial", "starter", "pro", "enterprise"],
    default: "trial"
  },
  createdAt: { type: Date, default: Date.now }
});

// Hash password before saving
userSchema.pre("save", async function(next) {
  if (!this.isModified("password")) return next();
  
  this.password = await bcrypt.hash(this.password, 10);
  next();
});

// Method to compare passwords
userSchema.methods.comparePassword = async function(password) {
  return await bcrypt.compare(password, this.password);
};

module.exports = mongoose.model("User", userSchema);
```

### Day 5: Authentication Routes

**backend/src/routes/auth.routes.js**
```javascript
const express = require("express");
const jwt = require("jsonwebtoken");
const User = require("../models/User");
const Business = require("../models/Business");

const router = express.Router();

// Signup
router.post("/signup", async (req, res) => {
  try {
    const { email, password, businessName } = req.body;
    
    // Validate
    if (!email || !password || !businessName) {
      return res.status(400).json({ error: "Missing required fields" });
    }
    
    // Check if user exists
    const existingUser = await User.findOne({ email });
    if (existingUser) {
      return res.status(400).json({ error: "User already exists" });
    }
    
    // Create user
    const user = new User({ email, password });
    await user.save();
    
    // Create business
    const business = new Business({
      name: businessName,
      industry: "dental", // Default, user can change
      ownerId: user._id
    });
    await business.save();
    
    // Link
    user.businessId = business._id;
    await user.save();
    
    // Generate token
    const token = jwt.sign(
      { userId: user._id, businessId: business._id },
      process.env.JWT_SECRET,
      { expiresIn: process.env.JWT_EXPIRE }
    );
    
    res.status(201).json({
      user: { id: user._id, email: user.email },
      business,
      token
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Login
router.post("/login", async (req, res) => {
  try {
    const { email, password } = req.body;
    
    const user = await User.findOne({ email });
    if (!user) {
      return res.status(401).json({ error: "Invalid credentials" });
    }
    
    const isValid = await user.comparePassword(password);
    if (!isValid) {
      return res.status(401).json({ error: "Invalid credentials" });
    }
    
    const token = jwt.sign(
      { userId: user._id, businessId: user.businessId },
      process.env.JWT_SECRET,
      { expiresIn: process.env.JWT_EXPIRE }
    );
    
    res.json({ user: { id: user._id, email: user.email }, token });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

module.exports = router;
```

### Day 6-7: Middleware & Testing

**backend/src/middleware/auth.middleware.js**
```javascript
const jwt = require("jsonwebtoken");

const verifyToken = (req, res, next) => {
  try {
    const token = req.headers.authorization?.split(" ")[1];
    
    if (!token) {
      return res.status(401).json({ error: "No token provided" });
    }
    
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.userId = decoded.userId;
    req.businessId = decoded.businessId;
    next();
  } catch (error) {
    res.status(401).json({ error: "Invalid token" });
  }
};

module.exports = { verifyToken };
```

### Day 1-2: Lead Service & API

**backend/src/services/LeadService.js**
```javascript
const Lead = require("../models/Lead");
const Business = require("../models/Business");

class LeadService {
  async createLead(businessId, leadData) {
    const lead = new Lead({
      businessId,
      firstName: leadData.firstName,
      lastName: leadData.lastName,
      email: leadData.email,
      phone: leadData.phone,
      source: leadData.source || "website_form",
      interestedService: leadData.interestedService,
      score: 10
    });
    
    await lead.save();
    return lead;
  }
  
  async scoreLead(leadId, action) {
    const lead = await Lead.findById(leadId);
    const business = await Business.findById(lead.businessId);
    
    const scoreRules = business.leadScoring?.rules || {
      chatbot_engaged: 20,
      form_submitted: 10,
      viewed_availability: 15
    };
    
    const points = scoreRules[action] || 5;
    lead.score += points;
    
    lead.scoreHistory.push({
      action,
      score: points,
      timestamp: new Date()
    });
    
    await lead.save();
    return lead;
  }
  
  async getLeads(businessId, filters = {}) {
    const query = { businessId };
    
    if (filters.status) query.status = filters.status;
    if (filters.source) query.source = filters.source;
    
    const leads = await Lead.find(query)
      .sort({ createdAt: -1 })
      .limit(filters.limit || 50)
      .skip(filters.skip || 0);
    
    const total = await Lead.countDocuments(query);
    
    return { leads, total };
  }
}

module.exports = new LeadService();
```

**backend/src/routes/leads.routes.js**
```javascript
const express = require("express");
const { verifyToken } = require("../middleware/auth.middleware");
const Lead = require("../models/Lead");
const LeadService = require("../services/LeadService");

const router = express.Router();

// Create lead (Public - for embed widget)
router.post("/", async (req, res) => {
  try {
    const { businessId, firstName, lastName, email, phone, interestedService } = req.body;
    
    const lead = await LeadService.createLead(businessId, {
      firstName,
      lastName,
      email,
      phone,
      interestedService,
      source: "website_form"
    });
    
    res.status(201).json({ lead });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Get leads (Protected)
router.get("/:businessId", verifyToken, async (req, res) => {
  try {
    const { status, source, limit = 50, skip = 0 } = req.query;
    
    // Verify user owns this business
    if (req.businessId !== req.params.businessId) {
      return res.status(403).json({ error: "Unauthorized" });
    }
    
    const { leads, total } = await LeadService.getLeads(req.params.businessId, {
      status,
      source,
      limit: parseInt(limit),
      skip: parseInt(skip)
    });
    
    res.json({ leads, total, pages: Math.ceil(total / limit) });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Score lead
router.post("/:businessId/:leadId/score", async (req, res) => {
  try {
    const { action } = req.body;
    const lead = await LeadService.scoreLead(req.params.leadId, action);
    res.json({ lead });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

module.exports = router;
```

```javascript
const Anthropic = require("@anthropic-ai/sdk");
const Conversation = require("../models/Conversation");
const Lead = require("../models/Lead");
const Business = require("../models/Business");
const LeadService = require("./LeadService");

class ChatbotService {
  constructor() {
    this.client = new Anthropic({
      apiKey: process.env.ANTHROPIC_API_KEY
    });
  }
  
  async generateResponse(businessId, leadId, userMessage) {
    // Get data
    const business = await Business.findById(businessId);
    const lead = await Lead.findById(leadId);
    let conversation = await Conversation.findOne({ businessId, leadId });
    
    if (!conversation) {
      conversation = new Conversation({
        businessId,
        leadId,
        conversationType: "chatbot",
        messages: []
      });
      await conversation.save();
    }
    
    // Build system prompt
    const systemPrompt = this.buildPrompt(business, lead);
    
    // Prepare messages for Claude
    const messages = conversation.messages.map(msg => ({
      role: msg.sender === "chatbot" ? "assistant" : "user",
      content: msg.content
    }));
    
    // Call Claude
    try {
      const response = await this.client.messages.create({
        model: "claude-3-5-sonnet-20241022",
        max_tokens: 300,
        system: systemPrompt,
        messages: [
          ...messages,
          { role: "user", content: userMessage }
        ]
      });
      
      const botMessage = response.content[0].text;
      
      // Save conversation
      conversation.messages.push({
        sender: "lead",
        content: userMessage,
        timestamp: new Date(),
        type: "text"
      });
      
      conversation.messages.push({
        sender: "chatbot",
        senderName: business.chatbotConfig?.name || "Assistant",
        content: botMessage,
        timestamp: new Date(),
        type: "text",
        isAiGenerated: true,
        confidence: 0.95,
        tokensUsed: response.usage.output_tokens
      });
      
      await conversation.save();
      
      // Update lead score
      await LeadService.scoreLead(leadId, "chatbot_engaged");
      
      return {
        message: botMessage,
        tokensUsed: response.usage.output_tokens
      };
      
    } catch (error) {
      console.error("Claude API error:", error);
      return {
        message: "Sorry, I couldn't process that. Please try again or contact us directly.",
        error: true
      };
    }
  }
  
  buildPrompt(business, lead) {
    return `You are a friendly booking assistant for ${business.name}.

Business Details:
- Type: ${business.industry}
- Services: ${business.services.map(s => `${s.name} ($${s.price})`).join(", ")}
- Hours: ${JSON.stringify(business.settings.workHours)}

Lead Info:
- Name: ${lead.firstName} ${lead.lastName}
- Phone: ${lead.phone}
- Email: ${lead.email}

Your Goals:
1. Be warm and professional
2. Understand their service needs
3. Suggest appointment times
4. Confirm booking

Important: Be concise (1-2 sentences), ask one question at a time.`;
  }
}

module.exports = new ChatbotService();
```

```javascript
const Appointment = require("../models/Appointment");
const Lead = require("../models/Lead");
const Business = require("../models/Business");
const GoogleCalendarService = require("./GoogleCalendarService");
const NotificationService = require("./NotificationService");

class AppointmentService {
  async createAppointment(businessId, leadId, appointmentData) {
    const appointment = new Appointment({
      businessId,
      leadId,
      serviceName: appointmentData.serviceName,
      startTime: appointmentData.startTime,
      endTime: appointmentData.endTime,
      status: "confirmed"
    });
    
    await appointment.save();
    
    // Update lead
    const lead = await Lead.findByIdAndUpdate(
      leadId,
      {
        appointmentId: appointment._id,
        status: "converted"
      },
      { new: true }
    );
    
    // Sync to Google Calendar
    try {
      await GoogleCalendarService.syncEvent(businessId, appointment._id);
    } catch (error) {
      console.error("Calendar sync error:", error);
    }
    
    // Send confirmation
    await NotificationService.sendAppointmentConfirmation(
      businessId,
      appointment
    );
    
    // Schedule reminders
    await this.scheduleReminders(appointment._id);
    
    return appointment;
  }
  
  async scheduleReminders(appointmentId) {
    const appointment = await Appointment.findById(appointmentId);
    
    // 24h reminder
    const reminder24h = new Date(appointment.startTime);
    reminder24h.setHours(reminder24h.getHours() - 24);
    
    // Schedule job (would use Bull queue in production)
    // For now, just log
    console.log(`Scheduled 24h reminder for ${appointmentId}`);
  }
}
```
